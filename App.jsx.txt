import { useState, useRef, useEffect } from "react";

const API_BASE = "http://localhost:8000";

const TOPICS = ["General", "Python", "Data Structures", "Algorithms", "System Design", "Machine Learning", "SQL", "OOP", "Behavioral", "GenAI/LLM"];

const SAMPLE_QUESTIONS = [
  "What is the difference between list and tuple in Python?",
  "Explain RAG architecture",
  "Tell me about yourself",
  "What is overfitting?",
  "Explain REST API",
];

function MarkdownAnswer({ text }) {
  const formatted = text
    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
    .replace(/\n/g, '<br/>');
  return <div dangerouslySetInnerHTML={{ __html: formatted }} />;
}

export default function App() {
  const [question, setQuestion] = useState("");
  const [topic, setTopic] = useState("General");
  const [messages, setMessages] = useState([
    {
      role: "bot",
      text: "👋 Hi Bala! I'm your Interview Prep Coach. Ask me any interview question — technical or behavioral. I'll give you a structured answer with examples and follow-ups!",
    },
  ]);
  const [loading, setLoading] = useState(false);
  const [uploadText, setUploadText] = useState("");
  const [uploadName, setUploadName] = useState("");
  const [uploadStatus, setUploadStatus] = useState("");
  const [tab, setTab] = useState("chat");
  const bottomRef = useRef(null);

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  async function handleAsk() {
    if (!question.trim()) return;
    const q = question.trim();
    setQuestion("");
    setMessages((prev) => [...prev, { role: "user", text: q }]);
    setLoading(true);

    try {
      const res = await fetch(`${API_BASE}/ask`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ question: q, topic }),
      });
      const data = await res.json();
      setMessages((prev) => [
        ...prev,
        {
          role: "bot",
          text: data.answer,
          sources: data.sources,
          context_used: data.context_used,
        },
      ]);
    } catch {
      setMessages((prev) => [
        ...prev,
        { role: "bot", text: "❌ Could not connect to backend. Make sure the FastAPI server is running on port 8000." },
      ]);
    }
    setLoading(false);
  }

  async function handleUpload() {
    if (!uploadText.trim() || !uploadName.trim()) return;
    setUploadStatus("Uploading...");
    try {
      const res = await fetch(`${API_BASE}/upload`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ content: uploadText, filename: uploadName }),
      });
      const data = await res.json();
      setUploadStatus(`✅ ${data.message}`);
      setUploadText("");
      setUploadName("");
    } catch {
      setUploadStatus("❌ Upload failed. Is the backend running?");
    }
  }

  return (
    <div style={{
      fontFamily: "'Inter', sans-serif",
      background: "#0f1117",
      minHeight: "100vh",
      color: "#e2e8f0",
      display: "flex",
      flexDirection: "column",
      alignItems: "center",
    }}>
      {/* Header */}
      <div style={{
        width: "100%",
        background: "linear-gradient(135deg, #1a1f2e 0%, #16213e 100%)",
        borderBottom: "1px solid #2d3748",
        padding: "16px 24px",
        display: "flex",
        alignItems: "center",
        gap: "12px",
        boxSizing: "border-box",
      }}>
        <div style={{
          width: 40, height: 40, borderRadius: "10px",
          background: "linear-gradient(135deg, #667eea, #764ba2)",
          display: "flex", alignItems: "center", justifyContent: "center",
          fontSize: "20px"
        }}>🤖</div>
        <div>
          <div style={{ fontWeight: 700, fontSize: "18px", color: "#fff" }}>Interview Prep Bot</div>
          <div style={{ fontSize: "12px", color: "#7c86a2" }}>RAG + Claude AI · Built by Bala</div>
        </div>
        <div style={{
          marginLeft: "auto", background: "#1e2a3a", border: "1px solid #2d3748",
          borderRadius: "8px", padding: "4px 12px", fontSize: "12px", color: "#48bb78"
        }}>
          ● Live
        </div>
      </div>

      {/* Tabs */}
      <div style={{
        display: "flex", gap: "0", marginTop: "20px",
        background: "#1a1f2e", borderRadius: "10px", padding: "4px",
        border: "1px solid #2d3748"
      }}>
        {["chat", "upload"].map((t) => (
          <button key={t} onClick={() => setTab(t)} style={{
            padding: "8px 24px", borderRadius: "8px", border: "none",
            background: tab === t ? "linear-gradient(135deg, #667eea, #764ba2)" : "transparent",
            color: tab === t ? "#fff" : "#7c86a2",
            fontWeight: tab === t ? 600 : 400,
            cursor: "pointer", fontSize: "14px", transition: "all 0.2s"
          }}>
            {t === "chat" ? "💬 Chat" : "📄 Upload Docs"}
          </button>
        ))}
      </div>

      <div style={{ width: "100%", maxWidth: "760px", padding: "0 16px", boxSizing: "border-box", flex: 1, display: "flex", flexDirection: "column" }}>

        {tab === "chat" && (
          <>
            {/* Topic selector */}
            <div style={{ display: "flex", gap: "8px", flexWrap: "wrap", marginTop: "16px" }}>
              {TOPICS.map((t) => (
                <button key={t} onClick={() => setTopic(t)} style={{
                  padding: "4px 12px", borderRadius: "20px", border: "none",
                  background: topic === t ? "#667eea" : "#1a1f2e",
                  color: topic === t ? "#fff" : "#7c86a2",
                  fontSize: "12px", cursor: "pointer",
                  border: topic === t ? "none" : "1px solid #2d3748",
                  transition: "all 0.2s"
                }}>{t}</button>
              ))}
            </div>

            {/* Sample questions */}
            <div style={{ marginTop: "12px", display: "flex", gap: "8px", flexWrap: "wrap" }}>
              {SAMPLE_QUESTIONS.map((q) => (
                <button key={q} onClick={() => setQuestion(q)} style={{
                  padding: "4px 10px", borderRadius: "6px",
                  background: "#1e2a3a", border: "1px solid #2d3748",
                  color: "#94a3b8", fontSize: "11px", cursor: "pointer"
                }}>✨ {q.slice(0, 35)}...</button>
              ))}
            </div>

            {/* Messages */}
            <div style={{
              flex: 1, overflowY: "auto", marginTop: "16px",
              display: "flex", flexDirection: "column", gap: "16px",
              maxHeight: "480px", paddingRight: "4px"
            }}>
              {messages.map((msg, i) => (
                <div key={i} style={{
                  display: "flex", justifyContent: msg.role === "user" ? "flex-end" : "flex-start"
                }}>
                  {msg.role === "bot" && (
                    <div style={{
                      width: 32, height: 32, borderRadius: "8px", flexShrink: 0,
                      background: "linear-gradient(135deg, #667eea, #764ba2)",
                      display: "flex", alignItems: "center", justifyContent: "center",
                      marginRight: "10px", fontSize: "14px"
                    }}>🤖</div>
                  )}
                  <div style={{
                    maxWidth: "82%",
                    background: msg.role === "user"
                      ? "linear-gradient(135deg, #667eea, #764ba2)"
                      : "#1a1f2e",
                    border: msg.role === "bot" ? "1px solid #2d3748" : "none",
                    borderRadius: msg.role === "user" ? "18px 18px 4px 18px" : "4px 18px 18px 18px",
                    padding: "12px 16px",
                    fontSize: "14px",
                    lineHeight: "1.6",
                    color: "#e2e8f0"
                  }}>
                    {msg.role === "bot" ? <MarkdownAnswer text={msg.text} /> : msg.text}
                    {msg.sources && (
                      <div style={{ marginTop: "8px", display: "flex", gap: "6px", flexWrap: "wrap" }}>
                        {[...new Set(msg.sources)].map((s, j) => (
                          <span key={j} style={{
                            background: "#0f1117", border: "1px solid #2d3748",
                            borderRadius: "4px", padding: "2px 8px", fontSize: "11px", color: "#667eea"
                          }}>📚 {s}</span>
                        ))}
                      </div>
                    )}
                  </div>
                </div>
              ))}
              {loading && (
                <div style={{ display: "flex", alignItems: "center", gap: "10px" }}>
                  <div style={{
                    width: 32, height: 32, borderRadius: "8px",
                    background: "linear-gradient(135deg, #667eea, #764ba2)",
                    display: "flex", alignItems: "center", justifyContent: "center", fontSize: "14px"
                  }}>🤖</div>
                  <div style={{
                    background: "#1a1f2e", border: "1px solid #2d3748",
                    borderRadius: "4px 18px 18px 18px", padding: "12px 20px",
                    fontSize: "14px", color: "#667eea"
                  }}>
                    Thinking<span style={{ animation: "blink 1s infinite" }}>...</span>
                  </div>
                </div>
              )}
              <div ref={bottomRef} />
            </div>

            {/* Input */}
            <div style={{
              display: "flex", gap: "8px", marginTop: "16px", marginBottom: "20px",
              background: "#1a1f2e", border: "1px solid #2d3748",
              borderRadius: "12px", padding: "8px"
            }}>
              <input
                value={question}
                onChange={(e) => setQuestion(e.target.value)}
                onKeyDown={(e) => e.key === "Enter" && !loading && handleAsk()}
                placeholder={`Ask a ${topic} interview question...`}
                style={{
                  flex: 1, background: "transparent", border: "none",
                  color: "#e2e8f0", fontSize: "14px", outline: "none", padding: "6px 8px"
                }}
              />
              <button onClick={handleAsk} disabled={loading || !question.trim()} style={{
                background: loading ? "#2d3748" : "linear-gradient(135deg, #667eea, #764ba2)",
                border: "none", borderRadius: "8px", color: "#fff",
                padding: "8px 20px", cursor: loading ? "not-allowed" : "pointer",
                fontWeight: 600, fontSize: "14px"
              }}>
                {loading ? "..." : "Ask →"}
              </button>
            </div>
          </>
        )}

        {tab === "upload" && (
          <div style={{ marginTop: "20px" }}>
            <div style={{
              background: "#1a1f2e", border: "1px solid #2d3748",
              borderRadius: "12px", padding: "20px"
            }}>
              <h3 style={{ margin: "0 0 8px 0", color: "#667eea" }}>📄 Add Custom Knowledge</h3>
              <p style={{ margin: "0 0 16px 0", color: "#7c86a2", fontSize: "13px" }}>
                Paste your notes, resume, job description, or any study material. The bot will use it to answer your questions.
              </p>

              <input
                value={uploadName}
                onChange={(e) => setUploadName(e.target.value)}
                placeholder="Document name (e.g. My Resume, DSA Notes)"
                style={{
                  width: "100%", background: "#0f1117", border: "1px solid #2d3748",
                  borderRadius: "8px", padding: "10px 12px", color: "#e2e8f0",
                  fontSize: "13px", boxSizing: "border-box", marginBottom: "12px"
                }}
              />
              <textarea
                value={uploadText}
                onChange={(e) => setUploadText(e.target.value)}
                placeholder="Paste your content here..."
                rows={10}
                style={{
                  width: "100%", background: "#0f1117", border: "1px solid #2d3748",
                  borderRadius: "8px", padding: "10px 12px", color: "#e2e8f0",
                  fontSize: "13px", boxSizing: "border-box", resize: "vertical",
                  fontFamily: "monospace", marginBottom: "12px"
                }}
              />
              <button onClick={handleUpload} disabled={!uploadText.trim() || !uploadName.trim()} style={{
                background: "linear-gradient(135deg, #667eea, #764ba2)",
                border: "none", borderRadius: "8px", color: "#fff",
                padding: "10px 24px", cursor: "pointer", fontWeight: 600, fontSize: "14px"
              }}>
                Upload to Knowledge Base
              </button>
              {uploadStatus && (
                <div style={{ marginTop: "12px", fontSize: "13px", color: uploadStatus.includes("✅") ? "#48bb78" : "#fc8181" }}>
                  {uploadStatus}
                </div>
              )}
            </div>
          </div>
        )}
      </div>

      <style>{`
        @keyframes blink { 0%, 100% { opacity: 1; } 50% { opacity: 0.3; } }
        * { scrollbar-width: thin; scrollbar-color: #2d3748 transparent; }
        input::placeholder { color: #4a5568; }
        textarea::placeholder { color: #4a5568; }
      `}</style>
    </div>
  );
}
